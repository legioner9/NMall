     int fdrd,fdwt;

     main(argc,argv)
          int argc;
          char *argv[];

          if (argc != 3)
              exit(1);
          fork();

          if ((fdrd = open(argv[1],O_RDONLY)) == -1)
              exit(1);

                       ((fdwt = open(argv[2],O_WRONLY)) == -1))

          rdwrt();

     rdwrt()
     {
          for (;;)
          {
              if (read(fdrd,&c,1) != 1)
                       return;
              write(fdwt,&c,1);
          }
     }
